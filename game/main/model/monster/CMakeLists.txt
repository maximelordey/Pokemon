"${CMAKE_CURRENT_SOURCE_DIR}/include/iv.h"
"${CMAKE_CURRENT_SOURCE_DIR}/include/level.h"
"${CMAKE_CURRENT_SOURCE_DIR}/include/pokemon.h"
)

target_link_libraries(monster PUBLIC common experience specie)
